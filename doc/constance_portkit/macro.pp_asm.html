macro_rules! pp_asm {
    (
        @internal,
        unprocessed: [{ , $($unprocessed:tt)* }],
        code: [{ $($code:tt)* }],
    ) => { ... };
    (
        @internal,
        unprocessed: [{}],
        code: [{ $($code:tt)* }],
    ) => { ... };
    (
        @internal,
        unprocessed: [{ $fragment:tt $($unprocessed:tt)* }],
        code: [{ $($code:tt)* }],
    ) => { ... };
    (
        @done,
        unprocessed: [{ $($unprocessed:tt)* }],
        code: [{ $($code:tt)* }],
    ) => { ... };
    (
        // TODO: remove `$l:lit`
        $l:literal $($rest:tt)*
    ) => { ... };
}

Examples

#![feature(asm)]
#![feature(decl_macro)]

// miri doesn't support inline assembly
#[cfg(not(miri))]
unsafe {
    let output: usize;
    constance_portkit::pptext::pp_asm!(
        // The input fragments are simply concatenated. This means `mov`
        // would be part of this line comment if it didn't include a line
        // break.
        "# hoge \n"
        if cfg!(target_arch = "x86_64") {
            "mov {}, 42"
        }
        if cfg!(any(target_arch = "aarch64", target_arch = "arm")) {
            "mov {}, #42"
        }
        if cfg!(any(target_arch = "riscv32", target_arch = "riscv64")) {
            "li {}, 42"
        },
        out(reg) output
    );
    assert_eq!(output, 42);
}
